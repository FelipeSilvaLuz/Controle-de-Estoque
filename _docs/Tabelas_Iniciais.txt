create database ControleEstoque;
use ControleEstoque;

CREATE TABLE AutenticacaoUsuarios (
		UsuarioId		INT AUTO_INCREMENT PRIMARY KEY
        ,	Nome			VARCHAR(80)
        ,	CPF				VARCHAR(15)
        ,	Ramal			VARCHAR(10)
        ,	Email			VARCHAR(80)
        ,	Senha			VARCHAR(40)
        ,	CriadoEm		DateTime
        ,	CriadoPor		VARCHAR(80)
        ,	AlteradoEm		DateTime
        ,	AlteradoPor		VARCHAR(80)
        );


CREATE TABLE Produtos (
		ProdutoId		INT AUTO_INCREMENT PRIMARY KEY
        ,	Codigo			VARCHAR(50) 
        ,	Nome			VARCHAR(80)
        ,	Descricao		VARCHAR(200)
        ,	PrecoCusto		double
        ,	PrecoVenda		double
        ,   Quantidade      double
        ,	FotoBase64		varchar(10000)
        ,	NomeFoto		varchar(50)
        ,	Observacao		varchar(300)
        ,	CriadoEm		DateTime
        ,	CriadoPor		VARCHAR(80)
        ,	AlteradoEm		DateTime
        ,	AlteradoPor		VARCHAR(80)
        );
        ALTER TABLE Produtos ADD KEY (Codigo); 


 CREATE TABLE RegistrosVenda (
                          VendaId         int AUTO_INCREMENT PRIMARY KEY,
                          Codigo       VARCHAR(50),
                          NomeProduto     VARCHAR(200),
                          Vendedor 			VARCHAR(80),
                          PreçoUnitario      double,
                         Quantidade      int,
                         CriadoEm        datetime,
                          CriadoPor       VARCHAR(80),
                          AlteradoEm      datetime,
                          AlteradoPor     VARCHAR(80),
                          Status          VARCHAR(1),
                          FOREIGN KEY(Codigo) REFERENCES Produtos(Codigo)
                );
